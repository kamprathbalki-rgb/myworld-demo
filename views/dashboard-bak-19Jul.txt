<script>

function toggleFloatMenu(menuId){

    const menus =
    document.querySelectorAll(
        '.floater-menu'
    )

    menus.forEach(function(menu){

        if(menu.id !== menuId){
            menu.classList.remove('show')
        }

    })


    const selectedMenu =
    document.getElementById(menuId)


    selectedMenu.classList.toggle('show')

}

</script>
