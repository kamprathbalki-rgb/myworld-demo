const mongoose = require('mongoose')

const propertySchema = new mongoose.Schema({

    /*
    MASTER PROJECT RECORD
    One Project = One Record
    Example:
    Godrej Greens
    */

    tenantId: {
        type: mongoose.Schema.Types.ObjectId,
        ref: 'Tenant'
    },

    // Basic Project Information

    projectName: String,

    builderName: String,

    ownerName: String,

    ownerMobile: String,

propertyStatus: {
    type: String,
    default: 'Available'
},

floorNumber: Number,

totalFloors: Number,

facing: String,

liftAvailable: String,

propertyAge: Number,

coverPhoto: String,

singleFlatType: String,

singleCarpetArea: Number,

singleQuotedPrice: Number,

singleClosingPrice: Number,

askingPricePerSqFt: Number,

parkingType: String,

furnishedStatus: String,

possessionStatus: String,

    propertyMode: {
    type: String,
    enum: ['PROJECT', 'SINGLE'],
    default: 'PROJECT'
},

    projectType: {
        type: [String],
        enum: [
            'Residential',
            'Commercial',
            'Mixed Use'
        ],
        default: []
    },

    propertyType: {
        type: [String],
        enum: [
            'Apartment',
            'Villa',
            'Plot',
            'Office',
            'Showroom',
            'Retail',
            'Shop'
        ],
        default: []
    },

    projectStatus: {
        type: String,
        enum: [
            'New Launch',
            'Under Construction',
            'Ready Possession',
            'Resale'
        ]
    },

    // Simplified RERA status

    reraApproved: {
        type: String,
        enum: [
            'Yes',
            'No'
        ]
    },

    launchDate: Date,

    // Location Information

    city: String,

    direction: {
        type: String,
        enum: [
            'East',
            'West',
            'North',
            'South',
            'Central'
        ]
    },

    /*
    User selects only propertyLocation (officeName)

    These are auto-filled from LocationMaster:
    - divisionName
    - pincode
    - district
    - stateName
    */

    propertyLocation: String,   // officeName

    divisionName: String,

    pincode: Number,

    district: String,

    stateName: String,

    locationLandmark: String,

    googlePin: String,

    location: {
        type: {
            type: String,
            default: "Point"
        },
        coordinates: [Number]
    },

    // Towers / Buildings

    numberOfTowers: Number,

    towers: [
        {
            towerName: String,

            numberOfFloors: Number,

            towerStatus: {
                type: String,
                enum: [
                    'Under Construction',
                    'Ready Possession',
                    'Sold Out'
                ]
            },

            inventoryStatus: {
                type: String,
                enum: [
                    'Available',
                    'Limited Availability',
                    'Sold Out'
                ]
            },

            possessionMonth: String,

            possessionYear: Number
        }
    ],

    // Flat Configurations
    // One configuration = One flat type

    configurations: [
        {
            flatType: {
                type: String,
                enum: [
                    'Studio',
                    '1 RK',
                    '1 BHK',
                    '1.5 BHK',
                    '2 BHK',
                    '2.5 BHK',
                    '3 BHK',
                    '3.5 BHK',
                    '4 BHK',
                    '4.5 BHK',
                    '5 BHK',
                    '5+ BHK'
                ]
            },

            carpetArea: Number,

            quotedPrice: Number,

            closingPrice: Number,

            furnishedStatus: {
                type: String,
                enum: [
                    'Unfurnished',
                    'Semi-Furnished',
                    'Fully Furnished'
                ]
            },

            parkingType: {
                type: [String],
                enum: [
                    'Open',
                    'Covered',
                    'Basement',
                    'Podium',
                    'Stilt'
                ],
                default: []
            },

            possessionStatus: {
                type: [String],
                enum: [
                    'Immediate',
                    'Ready Possession',
                    'Under Construction',
                    'Possession Soon'
                ],
                default: []
            },

            towerName: String,

            availableUnits: Number
        }
    ],

    // Amenities & Sales Intelligence

    amenities: [String],

    usp: String,

    notes: String,

    images: [String],

    createdAt: {
        type: Date,
        default: Date.now
    }

})

module.exports = mongoose.model('Property', propertySchema)
